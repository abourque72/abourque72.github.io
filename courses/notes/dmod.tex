\section{Mar 24}

D-modules are modules over rings of differential operators. There are three settings:

\begin{enumerate}
	\item $C^\infty$ case: We can consider differential operators on $\R^n$ of order $N$, which are expressions \begin{gather*}
		u = \sum_{i_1+\cdots+i_n\le N} a_{i_1,\dots,i_n}(x)\frac{\partial^N}{\partial x_1^{i_1}\cdots \partial x_n^{i_n}},
	\end{gather*} where the functions $a_\bullet(x)$ are smooth. 
	\item Complex analytic setting: Similar, but replace $\R$ with $\C$ and require the functions $a_\bullet(x)$ are holomorphic.
	\item Algebraic setting: Replace $\C$ with any field and require the functions $a_\bullet(x)$ to be polynomials. 
\end{enumerate}

We will work with smooth algebraic varieties over $\C$.

\begin{rmk}
	\begin{enumerate}
		\item If $X$ is singular, the ring $\mc D(X)$ of differential operators might not be noetherian.
		\item The theory of differential operators in positive characteristic is very different.
		\item We can effectively replace $\C$ by any algebraically closed field of characteristic zero.
	\end{enumerate}
\end{rmk}

Today $X$ is smooth and affine. $\mc O=\mc O(X)$ will denote the algebra of regular functions on $X$. We might not have coordinates on $X$, so we need to modify our definition of differential operators.\\

Differential operators of order 0 are just multiplication by some $f\in\mc O$, denoted $m_f$. Differential operators of order 1 are $\xi + m_f$, where $f\in\mc O$ and $\xi$ is a vector field on $X$, i.e. an element of $T_X=\Der(\mc O,\mc O)$, i.e. a derivation $\mc O\to \mc O$. 

\begin{defn}
	The algebra $\mc D = \mc D(X)$ of differential operators on $X$ is defined as the subalgebra of $\End_\C(\mc O)$ generated by the operators of order 0 and 1 defined above. 
\end{defn}

Note that $\mc O+T_X$ sits as a subspace in $\mc D$. \\

When $X=\C^n$, or more generally is an open subvariety of $\C^n$, then this definition agrees with the previous one using the induced coordinates from $\C^n$. 

\subsection{Applications of D-modules}

Fix $f\in\mc O$. Let $\Ann(f) = \{u\in \mc D\mid u(f)=0\}$. It is a left ideal in $\mc D$. We define $\mc D f $ to be $\mc D/\Ann(f)$, which is a $\mc D$-module. \\

We can define $\mc Df$ for some $f\notin \mc O$:

\begin{enumerate}
	\item Let $X=\C$ and $f=e^x$. Certainly $f$ is not regular, but its annihilator is generated by $\ddx - 1$, so we can define $\mc D e^x = \mc D/(\ddx-1)$. 
	\item Similarly we can define $\Ann(f)$ and hence $\mc Df$ for $X=\C$ and $f\in C^\infty(\R)$. 
	\item Let $f\in\R[x_1,\dots,x_n]$ and let $U=\{x\in\R^n\mid f(x)\ge0\}$. Fix $\vphi\in C^\infty(U)$ with compact support. Let $I_\phi(\lambda)=\int_U\vphi(x)f(x)^\lambda dx$ for $\lambda\in\C$. Note that the exponentiation is well-defined because $f(x)\ge 0$. The integral converges when the real part of $\lambda$ is positive. Then \begin{thm}
		The function $\lambda\mapsto I_\vphi(\lambda)$ extends to a meromorphic function on $\C$ with poles at a finite union of arithmetic progressions of the form $a,a-1,\dots$. 
	\end{thm} \begin{proof}[Sketch]
	Say $I_\vphi(\lambda)$ converges for $\lambda>N$ (at the start, we have $N=0$, but we proceed inductively). We compute \begin{gather*}
		\dd \lambda I_\vphi(\lambda) = \int_U \vphi(x)\log f(x) f(x)^\lambda dx,
	\end{gather*} which converges, so $I_\vphi(\lambda)$ is holomorphic when $\Real \lambda \ge N$. Our goal is to extend the function to $\Real \lambda \ge N-1$ inductively. We need: \begin{lem}[$b$-function]\label{lem:bfunction}
	There exists $u\in\mc D(\C^n)[s]$ and $b\in\C[s]$ such that $u(f^{s+1})=b(s)f^s$. 
	\end{lem} Assuming the lemma, we have \begin{gather*}
	I_\vphi(\lambda) = \int_U\vphi(x)f(x)^\lambda dx = \int_U \vphi(x) \frac1{b(\lambda)}u(f(x)^{\lambda+1})dx \\
	= \frac 1{b(\lambda)}\int_U(u^*(\vphi))(x)f(x)^{\lambda+1}dx,
	\end{gather*} where at the end we have used essentially integration by parts, and $u^*\in\mc D(\C^n)[s]$ is the formal adjoint of $u$. For $\Real \lambda >N-1$ we can extend $I_\vphi(\lambda)$ using the above computation. 
	\end{proof}
\end{enumerate} 

\begin{exam}[Examples of $b$-functions]
	\begin{enumerate}
		\item Let $f(x)=\sum x_i^2$ on $\C^n$. Then we can take $u=\sum\ppnxi 2i$ and $b(s)=4(s+1)(s+\frac n2)$. 
		\item Let $f(x)=\det(x)$ on $\C^{n^2}=M_n(\C)$. Then for $u=\det(\ppxi{ij})$ one can compute $u(f^s)$ using Capelli identities and find that $b(s)=\prod\limits_{1\le k\le n}(s+k)$. 
		\item If $f(x_1,\dots, x_n)=\prod\limits_{1\le i<j\le n}$, then the $b$-function is unknown. Can also generalize to arbitrary root systems.
		\item If one takes the square of the above function, then the $b$-function is known, but is very complicated. 
	\end{enumerate}
\end{exam}

\subsection{Poincar\'e-Birkhoff-Witt (PBW) for differential operators}

We continue to assume $X$ is smooth affine, and we write $\mc O,\mc D$ for $\mc O(X),\mc D(X)$. We let $T_X$ denote the space of vector fields. \\

For $\xi\in T_X$ and $f,g\in\mc O$, we have the Liebniz formula $\xi(fg)=f\xi(g) + \xi(f)g$. In particular, in terms of operators, $\xi m_f - m_f\xi = m_{\xi(f)}$. Also, for $\eta\in T_X$, we have $\xi\eta-\eta\xi = [\xi,\eta]$, where $[-,-]$ is the (a priori abstract) Lie bracket of vector fields. It follows that $\mc D_1 = \mc O + T_X$ is a Lie algebra. $\mc O$ is an abelian Lie ideal and $T_X$ is a Lie subalgebra. We have a short exact sequence $0\to\mc O\to \mc D_1\to T_X\to 0$. The formula $\xi m_f - m_f \xi = m_{\xi(f)}$ also shows that $\mc D_1$ is an $\mc O$-bimodule, and that the left and right actions agree on $\mc D_1/\mc O$. 

\begin{defn}
	The \textbf{order filtration} on $\mc D$ is given by $\mc D_0=\mc O$, $\mc D_1 = \mc O+T_X$, and $\mc D_n$ is the image of the composition map $\mc D_1\otimes_{\mc O}\cdots \otimes_{\mc O} \mc D_1\to \mc D$. 
\end{defn}

We have \begin{itemize}
	\item $\bigcup_n \mc D_n=\mc D$.
	\item $\mc D_n$ is an $\mc O$-bimodule.
	\item $\mc D_m\mc D_n\subset \mc D_{m+n}$. 
\end{itemize}

Let $\gr \mc D = \bigoplus (\mc D_n/\mc D_{n-1})$. Let $f\in \mc O$ and $\xi,\eta\in T_X$. Then $f\xi\equiv \xi f$ mod $\mc O$, so $\mc O=\mc D_0$ is a central subalgebra of $\gr\mc D$. Similarly, $\xi\eta \equiv \eta\xi$ mod $\mc D_1$, so $T_X=\mc D_1/\mc D_0$ is contained in $Z(\gr \mc D)$. We obtain a surjective graded algebra map $\mu:\Sym_{\mc O} T_X\to \gr \mc D$. 

\begin{thm}[PBW for $\mc D$]\label{thm:PBWforD}
	$\mu$ is an isomorphism.
\end{thm} 

\begin{proof}
	It suffices to show that $\mu$ is injective. \\

\subsection{An application to the PBW theorem for Lie algebras}

\subsection{Sheaves of differential operators}

Let $X$ be an irreducible affine. Then $\mc O(X)$ is a domain. For any nonzero $f\in\mc O(X)$, we have the basic open $X_f=\{x\in X\mid f(x)\ne 0\}$. It is affine with coordinate ring $\mc O(X)_f = \mc O(X)[\frac1f]$. \\

Now assume $X$ is not necessarily affine. Facts: \begin{enumerate}
	\item A sheaf $\mc E$ of $\mc O_X$-modules is quasicoherent if and only if for every open affine $U\subset X$ and nonzero $f\in\mc O(U)$, the map \begin{gather}\label{eq:qcoh}
		\Gamma(U,\mc E)_f := \mc O(U_f)\otimes_{\mc O(U)}\Gamma(U,\mc E)\to \Gamma(U_f,\mc E)
	\end{gather} is an isomorphism. In this case/equivalently, for all affine opens $V\subset U$, there is an isomorphism \begin{gather*}
	\mc O(V)\otimes_{\mc O(U)}\Gamma(U,\mc E)\to \Gamma(V,\mc E).
	\end{gather*} 
	\item If $X$ is affine, then the map \begin{gather*}

\subsection{Twisted differential operators (TDO)}

Let $\mc L$ be a line bundle on $X$. Let $\mc D_{X,\mc L}=\mc L\otimes_{\mc O_X}\mc D_X\otimes_{\mc O_X}\mc L^{-1}$. Here $\mc L^{-1}$ is the dual bundle to $\mc L$, sometimes denoted $\mc L^\vee$ or $\mc L^*$. 

\begin{claim}
	$\mc D_{X,L}$ has the structure of a sheaf of associative algebras.
\end{claim}

\begin{proof}
	Let $u_1,u_2$ be local sections of $\mc D_X$. Let $\alpha_1,\alpha_2$ be local sections of $\mc L$. Let $\beta_1,\beta_2$ be local sections of $\mc L^{-1}$. We define $(\alpha_1u_1\beta_1)(\alpha_2u_2\beta_2)$ to be $\alpha_1(u_1 m_{\pair{\beta_1}{\alpha_2}})u_2)\beta_2$. 
\end{proof}

There is also a natural action of $\mc D_{X,\mc L}$ on $\mc L$, given on local sections by $(\alpha_1u\beta)\alpha_2 = u(\pair{\beta}{\alpha_2})\alpha_1$. \\

Replacing $\mc D_X$ by $\mc D_{X,n}$ in the definition of $\mc D_{X,\mc L}$ gives a sheaf we call $\mc D_{X,\mc L,n}$, giving a PBW filtration on $\mc D_{X,\mc L}$. We have \begin{gather*}
	\gr \mc D_{X,\mc L}=\mc L\otimes_{\mc O_X}\gr\mc D_X\otimes_{\mc O_X}\mc L^{-1} = \gr \mc D_X,
\end{gather*} where we can cancel $\mc L$ and $\mc L^{-1}$ since $\gr\mc D_X$ is an ordinary sheaf of $\mc O_X$ modules, i.e. there is no distinction between left and right. 

\begin{defn}
	A $\mc D_X$-\textbf{module} is a quasicoherent sheaf $\mc M$ such that for all open $U\subset X$, we have an action \begin{gather*}
		\Gamma(U,\mc D_X)\times \Gamma(U,\mc M)\to \Gamma(U,\mc M).
	\end{gather*}
\end{defn}

Recall $\mc D_X=\mc U(\mc O_X+ T_X)/(f\otimes u = fu\forall f\in \mc O_X,u\in \mc D_{X,1}; 1_{\mc U}=1_{\mc O_X})$. Thus, a $\mc D_X$-module is the same as a quasicoherent sheaf $\mc M$ with a ``flat connection", i.e. maps $\nabla_\xi$ for $\xi\in T_X$ acting on local sections of $\mc M$ such that $\nabla_\xi(fm)=f\nabla_\xi m + \xi(f)m$ and $\nabla_{[\xi,\eta]}=[\nabla_\xi,\nabla_\eta]$. \\

For a $\mc D_X$-module $\mc M$, $\Gamma(X,\mc M)$ is a module over $\Gamma(X,\mc D_X)=\mc D(X)$. In particular, $\Gamma(X,-)$ gives a functor from $\mc D_X$-modules to $\mc D(X)$-modules. It has a left adjoint called the localization functor $\Loc$, given by $\Loc(M)=\mc D_X\otimes_{\mc D(X)}M$. In particular, for an affine open $U\subset X$, we have $\Gamma(U,\Loc(M))=\mc D(U)\otimes_{\mc D(X)}M$. \\

If $X$ is affine, $\Gamma(X,-)$ is an equivalence $\QC(X)\to \mc O(X)$-mod with inverse $M\mapsto \mc O_X\otimes_{\mc O(X)}M$. Using the map \ref{eq:qcoh}, one can show that this restricted map from $\mc D_X$-modules to $\mc D(X)$-modules is also an equivalence (when $X$ is affine).

\subsection{Beilinson-Bernstein localization theorem, the statement}

Let $G$ be a connected semisimple algebraic group with fixed Borel subgroup $B$. Let $\mc B=G/B$ be the flag variety. 

\begin{thm}[Beilinson-Bernstein Localization Theorem]\label{thm:bbloc}
	$\Gamma(\mc B,-):\Rmod{\mc D_{\mc B}}\to \Rmod{\mc D(\mc B)}$ is an equivalence with inverse $\Loc$.
\end{thm}

\begin{rmk}
	A similar statement holds for all partial flag varieties $G/P$ with $P$ parabolic. In particular, it holds for projective spaces. Historically, this statement was first shown for projective spaces in a much simpler way. 
\end{rmk}

\subsection{\'Etale coordinates}

Let $X$ be smooth affine. By Noether normalization, shrinking $X$ if necessary, one can construct an \'etale map $F:X\to \C^n$; in particular (which is all we care about from the \'etale condition) $d_xF:T_xX\xr\sim T_{F(x)}\C^n$ for all $x\in X$. Let $\p_i = (dF)^*(\ppxi i)$. Then the $\p_i$ form a basis of $TX$ with the property that $[\p_i,\p_j]=0$ for all $i,j$. Thus any differential operator on $X$ can be expressed in the form \begin{gather*}
	\sum_{i_1+\cdots+i_n\le k} f_{i_1,\dots,i_n}(x)\p_1^{i_1}\cdots\p_n^{i_n}
\end{gather*} for some $f_\bullet \in \mc O(X)$. Then one can work with differential operators in essentially the same way as with a vector space.

\begin{prop}

\end{prop}

\begin{proof}
	``To show something is locally free, it suffices to show that locally it is free" - Victor Ginzburg. \\
	
	Fix $x\in X$ and let $\mc O_x$ be the local ring at $x$ with maximal ideal $\mf m_x$. Let $D = \mc O_x\otimes_{\mc O_X}\mc D_X$ and let $M=\mc O_x\otimes_{\mc O_X}\mc M$. Let $\ol m_i$ be a $\C$-basis for $\ol M$, and pick lifts $m_i\in M$. There is a map $p:\mc O_x^n\to M$ given by $(f_i)\mapsto \sum f_im_i$. Nakayama's lemma implies $p$ is surjective. The proof is done once we show the following:
	
	\begin{claim}
		$p$ is injective.
	\end{claim}
	\begin{proof}[Proof of Claim]
		Assume $p$ is not injective. Define the order of a tuple $(f_i)$ to be the maximum $k\ge0$ such that each $f_i$ is in $\mf m_x^k$. Let $\nu$ be the minimum order of all tuples in $\ker p$, and take $(f_i)\in\ker p$ with order $\nu$. Choose \'etale coordinates around $x\in X$ so that we have operators $\p_i$. 
		
		\begin{claim}
			There is some $\p_i = \p$ such that $\p f_1 \in \mf m_x^{\nu-1}$ but not in $\mf m_x^\nu$.
		\end{claim} 
		
		We do not prove this claim. Assuming it, we have \begin{gather*}
			0 = \p\bigp{ \sum f_im_i } = \sum(\p f_i)m_i + \sum f_i(\p m_i).
		\end{gather*} By surjectivity of $p$, there are $a_{ij}\in \mc O_x$ such that $\p m_i = \sum a_{ij}m_j$. Thus \begin{gather*}
		0 = \sum(\p f_i)m_i + \sum f_ia_{ij}m_j.
		\end{gather*} The coefficient of $m_1$ is $\p f_1+\sum f_ia_{i1}$, which is in $\mf m_x^{\nu-1}$ but not in $\mf m_x^\nu$. Thus we have produced a relation with order less than $\nu-1$, a contradiction.
	\end{proof}
\end{proof}

\subsection{Review of connections}

Let $X$ be smooth, or holomorphic, or an algebraic variety. Let $\mc O = \mc O_X$ and $\Omega^i$ the sheaf of $i$-forms on $X$. For an $\mc O$-module $\mc M$, we can form $\mc M\otimes_{\mc O}\Omega^i$ to get $\mc M$-valued forms. We note $\mc Hom_{\mc O}(T_X,\mc M)\cong \mc M\otimes \Omega^1$. 

\begin{defn}
	A \textbf{connection} on $\mc M$ is a map $\nabla:\mc M\to\mc M\otimes_{\mc O}\Omega^1$ such that $\nabla(fm)=m\otimes df + f\nabla m$. 
\end{defn}

\begin{exam}
	If $\mc M=\mc O$, we can take $\nabla = d$. 
\end{exam}

Identifying $\mc Hom_{\mc O}(T_X,\mc M)\cong \mc M\otimes \Omega^1$, we can interpret $\nabla m$ as an element of $\mc Hom_{\mc O}(T_X,\mc M)$, denoted $\xi\mapsto \nabla_\xi m$. 

\begin{exam}

\end{exam}

\subsection{Pullback of D-modules}

Given $F:X\to Y$ a morphism of varieties, we claim that there is a pullback functor $F^*:\mc \Rmod{\mc D_Y}\to \Rmod{\mc D_X}$. At the level of $\mc O$-modules, it is exactly the $\mc O$-module pullback, i.e. $F^*\mc M = \mc O_X\otimes_{F^{-1}\mc O_Y}F^{-1}\mc M$, where $F^{-1}$ denotes the naive sheaf pullback. However, it is not obvious why this has a $\mc D_X$-module structure, which we now explain in multiple ways. \\

The module $\mc M$ induces a flat connection $\nabla:\mc M\to \mc M\otimes_{\mc O_Y}\Omega^1_Y$. Pulling back the connection and composing with $F^*\Omega^1_Y\to \Omega^1_X$ gives a connection $F^*\nabla:F^*\mc M\to F^*\mc M\otimes_{\mc O_X}F^* \Omega^1_Y\to F^*\mc M\otimes_{\mc O_X}\Omega^1_X$. This connection is flat and thus gives a $\mc D_X$-module structure on $\mc M$. \\

We can also explain the D-module structure in terms of the action of vector fields. The map $F$ induces $dF:T_X\to F^*T_Y$. Writing sections of $F^*\mc M$ as $f\otimes F^{-1}m$ (via the defining expression of $F^*\mc M$), we write the action of $\xi\in T_X$ as \begin{gather*}
	\xi(f\otimes F^{-1}m)=\xi(f)\otimes F^{-1}m + f\otimes dF(\xi)m.
\end{gather*}

\end{exam}

\subsection{Pushforward of D-modules along an open embedding}

Let $F:X\into Y$ be an open embedding. Then $F_*$ for D-modules is the same as $F_*$ for $\mc O$-modules, which is the same as $F_*$ (naive pushforward) for sheaves of vector spaces. Let $\mc M$ be a sheaf on $X$, and let $U\subset Y$ be open. Then $F_*\mc M(U)=\mc M(U\cap X)$. The $\mc D_Y(U)$-module structure on $F_*\mc M(U)$ is then given by restricting to $\mc D_Y(U\cap X)=\mc D_X(U\cap X)$. We note that (in this case) $F_*$ is left exact, but not exact in general.
